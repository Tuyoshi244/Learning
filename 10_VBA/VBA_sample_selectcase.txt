※注意：「変数が定義されていません」エラーが出る場合
　　　　　"Option Explicit"を外してください。

使い方：シートの適当なところにフォームコントロールのボタンを挿入し
　　　　下記プログラムを配置してください。


sub SelectCaseSample()

'変数を宣言とデータ型
	Dim intdata As String

'メッセージボックスで変数を代入
	intdata = InputBox("ここはどこ？")

	Select Case intdata

'分岐①
    	Case "栄"
        
		MsgBox "正解"

'分岐②
	Case "名古屋"
        
		MsgBox "ほしい"

'分岐③
	Case Else
	
		MsgBox "帰ってください"

	End Select

End sub
